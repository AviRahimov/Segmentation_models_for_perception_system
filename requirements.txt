# Core deep-learning runtime.
#
# RTX 5090 is Blackwell (compute capability sm_120). Stock PyTorch wheels
# on PyPI only target up to sm_90, so any GPU op fails with
#   "CUDA error: no kernel image is available for execution on the device"
# on a 5090. The CUDA-12.8 wheels published by PyTorch include sm_120
# (and still cover sm_60 ... sm_90, so the same pin runs on the
# Jetson AGX Orin's sm_87 too once you install the JetPack-native
# torch wheel - see README for the Jetson note).
--extra-index-url https://download.pytorch.org/whl/cu128

# Vision / IO
#
# opencv-python-headless avoids a Qt plugin conflict with PyQt5/PyQt6 (plain
# opencv-python bundles its own Qt platform plugins that silently overwrite
# headless's files in the shared cv2/ install dir, since both packages
# install into the same location — crashes run_player.py at Qt-plugin-load
# time, not at import time). BOTH `supervision` and `trackers` below declare
# a hard, undeclared-as-optional dependency on plain `opencv-python` (not
# headless) — confirmed by inspecting their metadata directly. There is no
# way to make pip skip that transitive dependency; it WILL reinstall plain
# opencv-python on any fresh `pip install -r requirements.txt`, silently
# corrupting the headless install. After any fresh install, always run:
#   pip uninstall -y opencv-python && \
#   pip install --force-reinstall --no-deps opencv-python-headless==4.10.0.84
# (verify: `ls .venv/lib/python3.12/site-packages/cv2/qt/plugins/platforms/`
# should report "No such file or directory" — if libqxcb.so is present,
# plain opencv-python won the race and the fix above needs re-running.)
opencv-python-headless==4.10.0.84
Pillow==11.0.0
